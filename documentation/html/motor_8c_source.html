78 int motor_read_encoder() {
79  // Setting !OE low to enable output of encoder
80  PIOD->PIO_CODR |= NOT_OE;
81 
82  // Setting SEL low to get high byte and extracting MSB
83  PIOD->PIO_CODR |= SEL;
84  _delay_us(20);
85  uint8_t msb = (PIOC->PIO_PDSR & ENCODER_DATA_MASK) >> DO0_IDX;
86 
87  // Setting SEL high to get low byte and extracting LSB
88  PIOD->PIO_SODR |= SEL;
89  _delay_us(20);
90  uint8_t lsb = (PIOC->PIO_PDSR & ENCODER_DATA_MASK) >> DO0_IDX;
91 
92  // Reset encoder
93  PIOD->PIO_CODR |= NOT_RST;
94  PIOD->PIO_SODR |= NOT_RST;
95 
96  // Disable output of encoder
97  PIOD->PIO_SODR |= NOT_OE;
98 
99  // Combine LSB and MSB
100  uint16_t encoder_data = ((msb << 8) | lsb);
101  if (encoder_data & (1 << 15)) {
102  return ((uint16_t) (~encoder_data) + 1);
103  }
104  return -encoder_data;
105 }
